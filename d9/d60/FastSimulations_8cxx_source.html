36 void NeuralFastSimulation::setInputOutputData()
37 {
38  for (size_t i = 0; i < mSession.GetInputCount(); ++i) {
39  mInputNames.push_back(mSession.GetInputName(i, mAllocator));
40  }
41  for (size_t i = 0; i < mSession.GetInputCount(); ++i) {
42  mInputShapes.emplace_back(mSession.GetInputTypeInfo(i).GetTensorTypeAndShapeInfo().GetShape());
43  }
44  for (size_t i = 0; i < mSession.GetOutputCount(); ++i) {
45  mOutputNames.push_back(mSession.GetOutputName(i, mAllocator));
46  }
47 
48  // Prevent negative values from being passed as tensor shape
49  // Which is no problem in python implementation of ONNX where -1 means that
50  // shape has to be figured out by library. In C++ this is illegal
51  for (auto& shape : mInputShapes) {
52  for (auto& elem : shape) {
53  elem = std::abs(elem);
54  }
55  }
56 }

97 std::optional<std::pair<std::vector<float>, std::vector<float>>> o2::zdc::fastsim::loadScales(const std::string& path)
98 {
99  std::fstream file(path, file.in);
100  if (!file.is_open()) {
101  return std::nullopt;
102  }
103 
104  auto means = parse_block(file, "#means");
105  if (means.size() == 0) {
106  return std::nullopt;
107  }
108 
109  auto scales = parse_block(file, "#scales");
110  if (scales.size() == 0) {
111  return std::nullopt;
112  }
113 
114  if (means.size() != scales.size()) {
115  return std::nullopt;
116  }
117  return std::make_pair(std::move(means), std::move(scales));
118 }
